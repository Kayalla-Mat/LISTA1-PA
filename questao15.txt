#include <stdio.h>
#include <stdlib.h>

// Função para ler os valores e apresentá-los em ordem crescente
void ordenarValores(float *valores, int n) {
    // Algoritmo de ordenação: Bubble Sort
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (valores[j] > valores[j + 1]) {
                // Troca os valores
                float temp = valores[j];
                valores[j] = valores[j + 1];
                valores[j + 1] = temp;
            }
        }
    }

    printf("Valores em ordem crescente:\n");
    for (int i = 0; i < n; i++) {
        printf("%.2f ", valores[i]);
    }
    printf("\n");
}

int main() {
    int n;
    float *valores;

    printf("Quantos valores deseja inserir? ");
    scanf("%d", &n);

    // Alocação dinâmica de memória para o array de valores
    valores = (float *)malloc(n * sizeof(float));
    if (valores == NULL) {
        printf("Erro de alocação de memória!\n");
        return 1;
    }

    printf("Digite %d valores:\n", n);
    for (int i = 0; i < n; i++) {
        scanf("%f", &valores[i]);
    }

    // Chama a função para ordenar os valores e apresentá-los
    ordenarValores(valores, n);

    // Liberando a memória alocada
    free(valores);

    return 0;
}
